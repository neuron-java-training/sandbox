16  public class ArrayDequeTest implements Test {

21  	@Override
22  	public void init() {
23  		this.szamok = new ArrayDeque<Integer>();
24  		Random rnd = new Random();
25  		for (int i = 0; i < size; i++) {
26  			szamok.add(rnd.nextInt(size));
27  		}
28  
29  	}
30  
31  	@Override
32  	public void get(int n) {
33  		int elem;
34  		for (int i = 0; i < n; i++) {
35  			elem = szamok.getFirst();
36  		}
37  
38  	}
39  
40  	@Override
41  	public void delete(int startIndex, int n) {
42  		for (int i = startIndex; i < startIndex + n; i++) {
43  			szamok.remove(i);
44  		}
45  
46  	}
47  
48  	@Override
49  	public void sort() {
50  	}

53  		super();

55  	}
