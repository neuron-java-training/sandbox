23  	@Override
24  	public void write(List<?> t) {
25  		ObjectOutputStream out = null;
26  		try {
27  			out = new ObjectOutputStream(new BufferedOutputStream(
28  					new FileOutputStream(dataFile)));
29  			for (int i = 0; i < t.size(); i++) {
30  				Object obj = t.get(i);
31  				out.writeObject(obj);
32  			}
33  
34  		} catch (Exception e) {
35  			e.printStackTrace();
36  		} finally {
37  			if (out != null) {
38  				try {
39  					out.close();
40  				} catch (IOException e) {
41  				}
42  			}
43  
44  		}
45  
46  	}
